matrix WorldViewProj;
matrix World;
float3 DirLight;
vector Color;

struct VS_INPUT
{
	float3 vPosition:POSITION;
	float3 vNormal:NORMAL;
};

struct VS_OUTPUT
{
	vector vClipPosition:POSITION;
	vector uvCoords:TEXCOORD;
	vector cColor:COLOR;
};

VS_OUTPUT Main(VS_INPUT input)
{
	VS_OUTPUT output = (VS_OUTPUT)0;

	output.vClipPosition = mul(input.vPosition,WorldViewProj);
	
	float3 normal = mul(input.vNormal,World);
	
	float s = ( dot(normal,DirLight) );

	output.uvCoords.x = s*s;
	output.uvCoords.y = 0.5f;

	output.cColor = Color;

	return output;
}
